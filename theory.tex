\section{Introduction}

We define an action description language for representing dynamic systems of class DS4 satisfying assumptions A1--A8.

\section{Signature}

A domain description $D$ consists of:
\begin{itemize}
    \item A finite set of fluents $\mathcal{F}$
    \item A finite set of actions $\mathcal{A}$
\end{itemize}

\section{Literals}

A \textbf{literal} is either a fluent $f \in \mathcal{F}$ or its negation $\neg f$.

\section{States}

A \textbf{state} $s$ is a consistent set of literals (i.e., it does not contain both $f$ and $\neg f$).
States may be \textbf{partial}: for some fluent $f$, neither $f$ nor $\neg f$ needs to belong to $s$.

\section{Action Description}

Each action $a \in \mathcal{A}$ is defined as:
\[
a = \langle Pre(a), Eff(a), \kappa(a) \rangle
\]
where:
\begin{itemize}
    \item $Pre(a)$ is a set of preconditions (set of literals, can be empty),
    \item $Eff(a)$ is the effect of the action (a literal or \(\emptyset\)),
    \item $\kappa(a) \geq 0$ is the base cost of the action.
\end{itemize}

\section{Transition Function}

We define a transition function:
\[
Result(s,a) = s'
\]

\subsection*{Case 1: Preconditions Hold}

If $Pre(a) \subseteq s$, then:
\[
Result(s,a) =
\begin{cases}
(s \setminus \{\neg l\}) \cup \{l\} & \text{if } Eff(a) = l \\
s & \text{if } Eff(a) = \emptyset
\end{cases}
\]

\subsection*{Case 2: Preconditions Do Not Hold}

If $Pre(a) \nsubseteq s$, then:
\[
Result(s,a) = s
\]

\paragraph{Inertia (A1).}
All fluents not affected by an action remain unchanged.

\section{Execution Cost}

The cost of executing an action $a$ in state $s$ is:
\[
ExecCost(a,s) =
\begin{cases}
\kappa(a) & \text{if } Pre(a) \subseteq s \text{ and } Eff(a) \neq \emptyset \\
0 & \text{otherwise}
\end{cases}
\]

\section{Programs}

A program is a sequence of actions:
\[
P = (a_1, a_2, \dots, a_n), \quad n \geq 0
\]
Execution is defined as:
\[
s_0 \xrightarrow{a_1} s_1 \xrightarrow{a_2} \dots \xrightarrow{a_n} s_n
\]
where $s_i = Result(s_{i-1}, a_i)$.

Due to partial initial states (A7), a program may correspond to multiple possible execution paths (A3).

\section{Cost of a Program}

The total cost of executing $P$ in $s_0$ is:
\[
Cost(P, s_0) = \sum_{i=1}^{n} ExecCost(a_i, s_{i-1})
\]

\section{Action language statements}
\subsection{Value statement}
\[\bar{f} ~\text{after}~ A_1, \ldots, A_n\]
Where \(\bar{f}\in F\) is a fluent and \(A_1, \ldots, A_n \in A\) are actions.\\
Abbreviation when \(n=0\): initially \(\bar{f}\)

\subsection{Effect statement}
\[A_1 ~\text{causes}~ \bar{f} ~\text{if}~ g_1 ,\ldots, g_k\]
Where \(\bar{f},g_1,\ldots,g_k \in F\) are fluents and \(A_1\in A\) is an action.\\
Abbreviation when \(k=0\): \(A\) causes \(\bar{f}\)

\subsection{Cost statement}
\[A_1 ~\text{costs}~ \kappa ~\text{if}~ g_1 ,\ldots, g_k\]
Where \(g_1,\ldots,g_k \in F\) are fluents, \(A_1\in A\) is an action and \(\kappa > 0\).
Cost is \(0\) when there is no cost statement whose preconditions are satisfied.\\
Abbreviation when \(k=0\): \(A\) costs \(\kappa\)

\section{Query Language statements}

\subsection*{Q1: Goal satisfaction}
\[\gamma ~\text{after}~ P ~\text{from}~ s_0\]
Where \(\gamma\) is the goal condition (a set of literals), \(P\) is a program (sequence of actions) and \(s_0\) is the state
from which the program is started.\\
Abbreviation: ``\(\gamma ~\text{after}~ P\)'' if the program is started in the initial state.

The query is true iff gamma is the subset of the state \(s_n\) reached at the end of the program.

\subsection*{Q2: Executability with Cost}
\[P ~\text{executable with cost}~ \kappa ~\text{from}~ s_0\]
Where \(\kappa > 0\), \(P\) is a program (sequence of actions) and \(s_0\) is the state
from which the program is started.\\
Abbreviation: ``\(P ~\text{executable with cost}~ \kappa\)'' if the program is started in the initial state.

The query is true iff:
\[
Cost(P, s_0) \leq \kappa
\]

Since all actions are executable in every state (A6), every program is executable; thus, this query checks whether the execution cost is no greater than $\kappa$.

\subsection*{Q3: Executability with exact Cost}
\[P ~\text{executable with exact cost}~ \kappa ~\text{from}~ s_0\]
The same as above, except it checks for equality. It's true iff:
\[
Cost(P, s_0) = \kappa
\]

\section{Partial Initial States}

A partial initial state represents a set of complete states (completions), each assigning a truth value to every fluent while remaining consistent with the given literals.
\\Execution of a program $P$ from a partial state $s_0$ induces a set of possible trajectories:
\[
s_0^i \xrightarrow{P} s_n^i
\]

\subsection*{Q1: Goal satisfaction (Skeptical Semantics)}

\[\gamma ~\text{after}~ P ~\text{from}~ s_0\]
iff:
\[
\forall i: \gamma \subseteq s_n^i
\]

\subsection*{Q2: Executability with Cost (Skeptical Semantics)}

\[P ~\text{executable with cost}~ \kappa ~\text{from}~ s_0\]
iff:
\[
\forall i: Cost(P, s_0^i) \leq \kappa
\]

\subsection*{Q3: Executability with exact Cost (Skeptical Semantics)}

\[P ~\text{executable with exact cost}~ \kappa ~\text{from}~ s_0\]
iff:
\[
\forall i: Cost(P, s_0^i) = \kappa
\]

\section{Example}

Let:
\[
\mathcal{F} = \{doorOpen, hasKey\}
\]
Define actions:

\paragraph{openDoor}
\[
Pre(openDoor) = \{hasKey\}
\]
\[
Eff(openDoor) = doorOpen
\]
\[
\kappa(openDoor) = 5
\]

\paragraph{knock}
\[
Pre(knock) = \emptyset
\]
\[
Eff(knock) = \emptyset
\]
\[
\kappa(knock) = 2
\]
Initial state (partial):
\[
s_0 = \{\neg doorOpen\}
\]
The fluent $hasKey$ is unknown.

\subsection*{Completions of $s_0$}

\[
s_0^1 = \{\neg doorOpen, hasKey\}
\]
\[
s_0^2 = \{\neg doorOpen, \neg hasKey\}
\]

\subsection*{Program}

\[
P = (openDoor)
\]

\subsection*{Execution}

\paragraph{Case 1: $s_0^1$}

Precondition holds, so:
\[
s_1^1 = \{doorOpen, hasKey\}
\]
Cost:
\[
Cost(P, s_0^1) = 5
\]

\paragraph{Case 2: $s_0^2$}

Precondition does not hold, so:
\[
s_1^2 = \{\neg doorOpen, \neg hasKey\}
\]
Cost:
\[
Cost(P, s_0^2) = 0
\]

\subsection*{Final States}

\[
\{s_1^1, s_1^2\}
\]

\subsection*{Query Results}

\paragraph{Q1: Goal Satisfaction}

\[
\{doorOpen\} ~\text{after}~ P
\]

is false because $doorOpen$ does not hold in all final states.

\paragraph{Q2 and Q3: Executability with Cost}

\[
P ~\text{executable with exact cost}~ 5
\]
is false since $\exists i \; Cost(P, s_0^i) \neq 5$. However
\[
P ~\text{executable with cost}~ 5
\]
is true, because \(openDoor\) cannot cost more than \(5\).

\[
P ~\text{executable with cost}~ 0
\]
is false since $\exists i \; Cost(P, s_0^i) > 0$.
