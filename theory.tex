\section*{Introduction}

We define an action description language for representing dynamic systems of class \(\mathbb{DS}4\)
satisfying assumptions A1--A8 in the task description.

\section{Action language syntax}
Here \(\actions\) is the set of all actions and \(\fluents\) is the set of all fluents.

\subsection{Value statement}
\[\bar{f} ~\text{after}~ A_1, \ldots, A_n\]
Where \(\bar{f}\in \fluents\) is a fluent and \(A_1, \ldots, A_n \in \actions\) are actions.\\
Abbreviation when \(n=0\): initially \(\bar{f}\)

\subsection{Effect statement}
\[A ~\text{causes}~ \bar{f} ~\text{if}~ g_1 ,\ldots, g_k\]
Where \(\bar{f},g_1,\ldots,g_k \in \fluents\) are fluents and \(A\in \actions\) is an action.\\
Abbreviation when \(k=0\): \(A_1\) causes \(\bar{f}\)

\subsection{Cost statement}
\[A ~\text{costs}~ \kappa\]
Where \(A\in \actions\) is an action and \(\kappa>0\).

\section{Action language semantics}
\noindent A \textbf{signature} \(\Upsilon\) consists of:
\begin{itemize}
    \item A finite set of fluents $\fluents$
    \item A finite set of actions $\actions$
\end{itemize}

\parr
A non-empty set \(D\) of value/effect/cost statements is called an \textbf{action domain}.

\parr
A \textbf{state} is a mapping \(\sigma:\fluents \rightarrow \{0,1\}\). If \(\sigma(f)=1\) we will
write it as \(\sigma\models f\) (\(f\) holds in \(\sigma\)). We will use the symbol \(\states\) for
the set of all states.

\parr
A \textbf{transition function} is a mapping \(\Psi: \actions\times\states\rightarrow\states\).
The value \(\Psi(A, \sigma)\) is the state resulting from performing action \(A\) in the state \(\sigma\).
In case of the \(\mathbb{DS}4\) class of dynamic systems the mapping \(\Psi\) must be defined for all
actions and states because performing an action is always possible.

\parr
A \textbf{cost function} is a mapping \(Cost: \actions\times\states\rightarrow\mathbb{N}\) such that:
\[Cost(A, \sigma) = \begin{cases}
\kappa & \text{if } (A ~\text{costs}~ \kappa) \in D ~\text{and}~ \Psi(A, \sigma)\neq\emptyset \\
0 & \text{otherwise}
\end{cases}\]
where \(D\) is the action domain. If multiple cost statements for a single action \(A\) are present
in \(D\), the domain is inconsistent and the cost function is undefined.

\parr
Let \(\mathcal{L}\) be an action language for a \(\mathbb{DS}4\) dynamic system. A \textbf{structure for }\(\mathcal{L}\)
is a triple \(S=(\Psi, \sigma_0, Cost)\), where:
\begin{itemize}
    \item \(\Psi\) is a transition function
    \item \(\sigma_0\in\states\) is the initial state
    \item \(Cost\) is the cost function
\end{itemize}

\parr A statement \(s\) \textbf{is true in structure} \(S\) (in symbols \(S\models s\)) iff:
\begin{itemize}
    \item \(s\) is of the form \[\bar{f} ~\text{after}~ A_1, \ldots, A_n\]
    and \(\Psi\star((A_1, \ldots, A_n),\sigma_0)\models \bar{f}\)
    \item \(s\) is of the form \[A ~\text{causes}~ \bar{f} ~\text{if}~ \bar{g_1} ,\ldots, \bar{g_k}\]
    and for every \(\sigma\in\states\) such that \(\sigma\models \bar{g_i}\): \(\Psi(A, \sigma)\models\bar{f}\)
    \item \(s\) is of the form \[A ~\text{costs}~ \kappa,\] function \(Cost\) is defined and
    \((\exists\sigma\in\states)~ Cost(A,\sigma)=\kappa\).
\end{itemize}

\parr A structure \(S=(\Psi, \sigma_0, Cost)\) is a \textbf{model} of action domain \(D\) in a language over the signature
\(\Upsilon=(\fluents,\actions)\) iff:
\begin{itemize}
    \item \((\forall s\in D) ~S\models s\)
    \item \(Cost\) exists and is defined for every \(A\in \actions\) and every \(\sigma\in\states\)
    \item for every effect statement \(s\) of the form \[A ~\text{causes}~ \bar{f} ~\text{if}~ g_1 ,\ldots, g_k\]
    if one of the following holds:
    \begin{itemize}
        \item \(s\in D\) and \((\exists i\in\{1,\ldots,k\})~ \sigma\nvDash \bar{g_i}\)
        \item \(s\notin D\)
    \end{itemize}
    then \(\sigma \models f\) iff \(\Psi(A,\sigma)\models f\) 
\end{itemize}

\parr A \textbf{program} \(P\) is a sequence of actions to be performed starting from the initial state.
The cost of a program \(P=(A_1,\ldots,A_k)\) is defined as:
\[Cost^\star((A_1,\ldots,A_k)) = \sum_{i=1}^k Cost(A_i, \Psi^\star((A_1,\ldots,A_i), \sigma_0))\]

\section{Query Language statements}

\subsection*{Q1: Goal satisfaction}
\[\gamma ~\text{after}~ P\]
Where \(\gamma\) is the goal condition (a set of literals) and \(P\) is a program.

\subsection*{Q2: Executability with Cost}
\[P ~\text{executable with cost}~ \kappa\]
Where \(\kappa > 0\) and \(P\) is a program.

\subsection*{Q3: Executability with exact Cost}
\[P ~\text{executable with exact cost}~ \kappa\]
Where \(\kappa > 0\) and \(P\) is a program.

\section{Semantics of the query language}

\subsection{Partial Initial States}

A partial initial state represents a set of complete states (completions), each assigning a truth value to every fluent while remaining consistent with the given literals.
\\Execution of a program $P$ from a partial state $s_0$ induces a set of possible trajectories:
\[
s_0^i \xrightarrow{P} s_n^i
\]

\subsection*{Q1: Goal satisfaction}
A query of the form:
\[\gamma ~\text{after}~ P\]
is a consequence of action domain \(D\) iff for every model \(S_i=(\Psi^i, \sigma_0^i, Cost)\):
\[
(\forall \bar{f}\in\gamma)~~ \Psi^i(P, \sigma_0^i) \models \bar{f}
\]

\subsection*{Q2: Executability with Cost}
A query of the form:
\[P ~\text{executable with cost}~ \kappa\]
is a consequence of action domain \(D\) iff for every model \(S_i=(\Psi^i, \sigma_0^i, Cost)\):
\[
Cost(P, \sigma_0^i) \leq \kappa
\]

\subsection*{Q3: Executability with exact Cost}
A query of the form:
\[P ~\text{executable with exact cost}~ \kappa\]
is a consequence of action domain \(D\) iff for every model \(S_i=(\Psi^i, \sigma_0^i, Cost)\):
\[
Cost(P, s_0^i) = \kappa
\]

\section{Example}

Let:
\[
\mathcal{F} = \{doorOpen, hasKey\}
\]

Let's consider the following statements:
\[\text{initially}~ \neg doorOpen\]
\[openDoor ~\text{causes}~ doorOpen ~\text{with cost}~ 5 ~\text{if}~ hasKey\]

\subsection*{Completions of $s_0$}

\[
s_0^1 = \{\neg doorOpen, hasKey\}
\]
\[
s_0^2 = \{\neg doorOpen, \neg hasKey\}
\]

\subsection*{Program}

\[
P = (openDoor)
\]

\subsection*{Execution}

\paragraph{Case 1: $s_0^1$}

Precondition holds, so:
\[
s_1^1 = \{doorOpen, hasKey\}
\]
Cost:
\[
Cost(P, s_0^1) = 5
\]

\paragraph{Case 2: $s_0^2$}

Precondition does not hold, so:
\[
s_1^2 = \{\neg doorOpen, \neg hasKey\}
\]
Cost:
\[
Cost(P, s_0^2) = 0
\]

\subsection*{Final States}

\[
\{s_1^1, s_1^2\}
\]

\subsection*{Query Results}

\paragraph{Q1: Goal Satisfaction}

\[
\{doorOpen\} ~\text{after}~ P
\]

is false because $doorOpen$ does not hold in all final states.

\paragraph{Q2 and Q3: Executability with Cost}

\[
P ~\text{executable with exact cost}~ 5
\]
is false since $\exists i \; Cost(P, s_0^i) \neq 5$. However
\[
P ~\text{executable with cost}~ 5
\]
is true, because \(openDoor\) cannot cost more than \(5\).

\[
P ~\text{executable with cost}~ 0
\]
is false since $\exists i \; Cost(P, s_0^i) > 0$.
